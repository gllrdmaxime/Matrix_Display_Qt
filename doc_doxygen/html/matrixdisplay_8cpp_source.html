125float MatrixDisplay::calculateCellSize() const
126{
127 if (MatrixRows <= 0)
128 return kMinCellSize;
129
130 int w = width();
131 int h = height();
132 if (w <= 0 || h <= 0)
133 return kMinCellSize;
134
135 float usableHeight = h * kMaxHeightUsage;
136 float sizeByHeight = usableHeight / static_cast<float>(MatrixRows);
137 float sizeByWidth = static_cast<float>(w) / static_cast<float>(MatrixCols);
138
139 float chosen = std::min(sizeByHeight, sizeByWidth);
140
141 if (chosen < kMinCellSize)
142 chosen = kMinCellSize;
143 if (kMaxCellSize > 0.0f && chosen > kMaxCellSize)
144 chosen = kMaxCellSize;
145
146 float dpr = devicePixelRatioF();
147 if (dpr > 0.0f) {
148 chosen = std::ceil(chosen * dpr) / dpr;
149 } else {
150 chosen = std::ceil(chosen);
151 }
152
153 return chosen;
154}
